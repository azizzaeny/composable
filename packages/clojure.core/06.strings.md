#### reMatches
```js path=dist/core.js

function reMatches(str, pattern) {
  var regex = new RegExp(pattern, "g");
  var matches = [];
  var match;
  while ((match = regex.exec(str)) !== null) {
    matches.push(match[0]);
  }
  return matches;
}
```
